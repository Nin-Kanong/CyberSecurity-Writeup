## Check Mac and IP addresses on Windows

### Check IP Address
````
import socket

def get_ip_address():
    try:
        # Get the hostname of the machine
        hostname = socket.gethostname()
        # Resolve the hostname to an IP address
        ip_address = socket.gethostbyname(hostname)
        print("===============================")
        print("   Checking IP Address (Windows)")
        print("===============================")
        print(f"Hostname: {hostname}")
        print(f"IP Address: {ip_address}")
        print("===============================")
    except Exception as e:
        print("Error retrieving IP address:", e)

if __name__ == "__main__":
    get_ip_address()
````

### Advance Check IP

````
import socket
import psutil

def get_basic_ip():
    hostname = socket.gethostname()
    ip_address = socket.gethostbyname(hostname)
    print("=== Basic Host Info ===")
    print(f"Hostname: {hostname}")
    print(f"Primary IP: {ip_address}")
    print()

def get_interface_details():
    print("=== Interface Details ===")
    for interface, addrs in psutil.net_if_addrs().items():
        print(f"Interface: {interface}")
        for addr in addrs:
            if addr.family == socket.AF_INET:  # IPv4
                print(f"  IPv4 Address : {addr.address}")
                print(f"  Netmask      : {addr.netmask}")
                print(f"  Broadcast    : {addr.broadcast}")
            elif addr.family == socket.AF_INET6:  # IPv6
                print(f"  IPv6 Address : {addr.address}")
                print(f"  Netmask      : {addr.netmask}")
                print(f"  Broadcast    : {addr.broadcast}")
        print()

def get_gateway_dns():
    print("=== Gateway & DNS Info ===")
    gws = psutil.net_if_stats()
    for iface, stats in gws.items():
        print(f"Interface: {iface}")
        print(f"  Status: {'Up' if stats.isup else 'Down'}")
        print(f"  MTU   : {stats.mtu}")
        print()
    # DNS servers (Windows reads from system config)
    import dns.resolver
    try:
        resolver = dns.resolver.Resolver()
        print("DNS Servers:")
        for server in resolver.nameservers:
            print(f"  {server}")
    except Exception as e:
        print("Could not retrieve DNS servers:", e)

if __name__ == "__main__":
    get_basic_ip()
    get_interface_details()
    get_gateway_dns()
````

### Check MAC Address 
````
import psutil

def get_mac_addresses():
    print("=== MAC Address Details ===")
    for interface, addrs in psutil.net_if_addrs().items():
        for addr in addrs:
            # AF_LINK is the MAC address family
            if addr.family == psutil.AF_LINK:
                print(f"Interface: {interface}")
                print(f"  MAC Address: {addr.address}")
                print()

if __name__ == "__main__":
    get_mac_addresses()
````
